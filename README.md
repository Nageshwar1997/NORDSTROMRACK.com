### Tailwind CSS

npm install -D tailwindcss
npx tailwindcss init

    "./src/**/*.{js,jsx,ts,tsx}",
  ],
  theme: {
    extend: {},
  },
  plugins: [],
}`

npm install react-router-dom

npm install react-icons --save
